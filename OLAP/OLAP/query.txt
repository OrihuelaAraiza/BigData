--QUERY PARA CREAR TABLA DEFACTS EN CLICKHOUSE

-- Tabla de Hechos (Fact Table)

CREATE TABLE clase_olap.fact_ventas (
    id_venta UInt64,
    fecha Date,
    id_producto UInt32,
    cantidad UInt32,
    total_venta Float64
) ENGINE = MergeTree() 
PARTITION BY toYYYYMM(fecha) 
ORDER BY (fecha, id_producto);

-- Dimensión Producto
CREATE TABLE clase_olap.dim_productos (
    id_producto UInt32,
    nombre String,
    categoria String,
    precio Float32
) ENGINE = ReplacingMergeTree() ORDER BY id_producto;

SELECT p.nombre, sum(v.cantidad) as total_vendido
FROM clase_olap.fact_ventas v
JOIN clase_olap.dim_productos p ON v.id_producto = p.id_producto
GROUP BY p.nombre
